def temprature(n, from)
	result = []
	from.downcase.strip!
	if from == "fahrenheit"
		c = ((5.0/9)*(n-32))
		k = ((5/9.0)*(n-32)+273.15)
		result[0] = c
		result[1] = k
	elsif from== "celcius"
		k = ((5/9.0)*(n-32)+273.15)
		f = (((n-273.15)*(9/5.0))+32)
		result[0] = k
		result[1] = f
	elsif from == "kelvin"
		c = ((5.0/9)*(n-32))
		f = (((n-273.15)*(9/5.0))+32)
		result[0] = c
		result[1] = f
	else
		puts from
		puts "Not found"
	end
	return result
	
end
